import matplotlib.pyplot as plt
import networkx as nx

def generate_void_instability(n):
    """
    จำลองการเกิด Von Neumann Ordinals: 0 -> {0} -> {0, {0}}
    """
    elements = []
    current_set = []
    
    for i in range(n):
        new_element = str(current_set)
        current_set.append(new_element)
        elements.append(list(current_set))
    return elements

def visualize_logic_growth(n):
    G = nx.DiGraph()
    G.add_node("Empty Set (∅)")
    
    prev_node = "Empty Set (∅)"
    for i in range(1, n + 1):
        curr_node = f"Ordinal {i}\n(Structure {i})"
        G.add_node(curr_node)
        G.add_edge(prev_node, curr_node)
        prev_node = curr_node
        
    pos = nx.spring_layout(G)
    nx.draw(G, pos, with_labels=True, node_color='lightblue', 
            node_size=3000, font_size=10, font_weight='bold', arrows=True)
    plt.title("Visualizing the Instability of the Void: Recursive Expansion")
    plt.show()

# Simulation of Logical Expansion
def void_instability_growth(iterations):
    universe = []
    current_state = set() # Start with empty
    for i in range(iterations):
        # Apply O(S) = S U {S}
        next_state = current_state.copy()
        next_state.add(frozenset(current_state))
        universe.append(next_state)

import numpy as np
import matplotlib.pyplot as plt

def calculate_logical_dimension(iterations):
    """
    คำนวณมิติ Hausdorff จากการเติบโตของเซตซ้อนเซต (Nested Sets)
    Scale (L): ระดับความลึกของการทำซ้ำ (Rank)
    Volume (N): จำนวนข้อมูลหรือความซับซ้อนที่อุบัติขึ้น
    """
    ranks = np.arange(1, iterations + 1)
    # สมมติฐาน: ข้อมูลเติบโตแบบก้าวกระโดดตามระดับความสัมพันธ์ (Power Set Growth)
    volumes = [2**i for i in range(iterations)] 
    
    # คำนวณมิติ D จาก log(Volume) / log(Scale)
    log_scales = np.log(ranks[1:])
    log_volumes = np.log(volumes[1:])
    
    # ใช้ Linear Regression หาค่าความชัน (Slope) ซึ่งคือค่ามิติ D
    coeffs = np.polyfit(log_scales, log_volumes, 1)
    dimension = coeffs[0]
    
    return ranks, volumes, dimension

# รันการจำลอง 10 ลำดับแรก
ranks, volumes, d_value = calculate_logical_dimension(10)

print(f"Calculated Structural Dimension: {d_value:.4f}")
